PASO 2 — Pega este código
const SHEET_ID = '1p_AcDoteeSGnnN6H3rr87n7Rt5hYimW7c9toqS4BUMg';

function doGet(e) {
  const sheet = e.parameter.sheet || '';
  const cb    = e.parameter.callback || 'cb';
  try {
    const ss  = SpreadsheetApp.openById(SHEET_ID);
    const tab = ss.getSheetByName(sheet);
    if (!tab) throw new Error('Hoja no encontrada: ' + sheet);
    const rows = tab.getDataRange().getValues();
    return out(cb, { ok: true, rows: rows });
  } catch(err) {
    return out(cb, { ok: false, error: err.message });
  }
}

function out(cb, data) {
  return ContentService
    .createTextOutput(cb + '(' + JSON.stringify(data) + ')')
    .setMimeType(ContentService.MimeType.JAVASCRIPT);
}
